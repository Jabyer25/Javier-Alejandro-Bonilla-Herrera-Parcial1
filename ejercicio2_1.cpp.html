#include <iostream>
using namespace std;

int main() {
    int a, suma;

    cout<<"Ingresar un numero menor a 50\n";
    cin>>a;

    if(a%3==0 &&a%5==0 && a <=50){
        for (int i = 0; i < a; ++i) {
            suma += i;

        }
        cout<< "Suma de numeros impares: "<<suma << endl;
    } else if (a%2==0 && a <=50){
        for (int i = 0; i < a; ++i) {
            suma += i;

        }
        cout<< "Suma de numeros pares: "<<suma << endl;
    }
    else  {
        cout<<"Ingreso un numero mayor a 50"<<endl;
    }

    return 0;
}
